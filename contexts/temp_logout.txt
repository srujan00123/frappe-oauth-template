  /**
   * Logout user
   */
  const logout = useCallback(() => {
    TokenStorage.clearAll();
    setIsAuthenticated(false);
    setUserInfo(null);
    setIdToken(null);
    setAccessToken(null);

    // Redirect to home page after logout
    if (typeof window !== "undefined") {
      window.location.href = "/";
    }
  }, []);
